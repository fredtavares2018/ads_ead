Cadastro simples - Estudando ->function<-

nome : type="text" id="nome"
email : type="text" id="email"
telefone : type="number" id="telefone"
button : type="submit" value="onClick=cadastro() -> Enviar" onClick="cadastro()"

function cadastro() {
let nome = document.getElementById('nome')
let email = document.getElementById('email')
let telefone = document.getElementById('telefone')
let table = document.getElementById('tablePesquisa')
let row = table.insertRow(0)
row.innerHTML = "" + nome.value + " " + email.value + " " + telefone.value + ""
}
